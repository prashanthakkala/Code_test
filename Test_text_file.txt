import pdfplumber
from lxml import etree

def pdf_to_html(pdf_path, output_html):
    # Open the PDF file
    with pdfplumber.open(pdf_path) as pdf:
        html = etree.Element("html")
        body = etree.SubElement(html, "body")
        
        # Extract content from each page
        for page in pdf.pages:
            page_div = etree.SubElement(body, "div")
            page_text = page.extract_text()
            
            # Add page content as paragraphs in the HTML body
            if page_text:
                for line in page_text.split("\n"):
                    p = etree.SubElement(page_div, "p")
                    p.text = line
        
        # Write HTML content to a file
        html_str = etree.tostring(html, pretty_print=True, method="html", encoding="unicode")
        with open(output_html, "w") as f:
            f.write(html_str)

# Example usage
pdf_to_html("example.pdf", "output.html")
