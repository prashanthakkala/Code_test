import pdfplumber

# Path to your PDF file
pdf_path = "your-file.pdf"

# Open the PDF file
with pdfplumber.open(pdf_path) as pdf:
    # Iterate over each page in the PDF
    for page_number, page in enumerate(pdf.pages, start=1):
        # Extract tables from the page
        tables = page.extract_tables()
        
        # Check if there are tables on the page
        if tables:
            print(f"Page {page_number} contains {len(tables)} table(s):")
            # Print each table found on the page
            for i, table in enumerate(tables, start=1):
                print(f"\nTable {i} on Page {page_number}:")
                for row in table:
                    print(row)
        else:
            print(f"Page {page_number} does not contain any tables.")
