import pdfplumber

# Path to your PDF file
pdf_path = "your-file.pdf"

# Open the PDF file
with pdfplumber.open(pdf_path) as pdf:
    # Iterate over each page in the PDF
    for page_number, page in enumerate(pdf.pages, start=1):
        # Extract tables from the page
        tables = page.extract_tables()
        
        # Check if there are tables on the page
        if tables:
            print(f"Page {page_number} contains {len(tables)} table(s):")
            # Print each table found on the page
            for i, table in enumerate(tables, start=1):
                print(f"\nTable {i} on Page {page_number}:")
                for row in table:
                    print(row)
        else:
            print(f"Page {page_number} does not contain any tables.")

import pdfplumber
import difflib

    # Iterate over each page in the PDF
    for page_number, page in enumerate(pdf.pages):
        print(f"Processing page {page_number + 1}")

        # Extract tables from the page
        tables = page.extract_tables()

        # Process each table on the page
        for table_number, table in enumerate(tables):
            print(f"Table {table_number + 1} on page {page_number + 1}")

            # Convert each row in the table to a string
            table_content = [' | '.join(row) for row in table]

            # Print the extracted table content
            print("\n".join(table_content))
            print("-" * 40)

            # If you want to compare the current table with the next one using difflib
            if table_number < len(tables) - 1:
                next_table_content = [' | '.join(row) for row in tables[table_number + 1]]

                diff = difflib.ndiff(table_content, next_table_content)

                # Print the differences
