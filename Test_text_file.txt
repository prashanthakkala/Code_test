from pdf2docx import Converter

# Path to the PDF file you want to convert
pdf_file = 'input.pdf'

# Path where you want to save the Word document
word_file = 'output.docx'

# Create a Converter object
converter = Converter(pdf_file)

# Convert the PDF to Word
converter.convert(word_file, start=0, end=None)

# Close the Converter object
converter.close()

print(f'Conversion completed! The Word file is saved as {word_file}')
