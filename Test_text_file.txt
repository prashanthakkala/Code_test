from pdf2docx import Converter

def convert_pdf_to_word(pdf_file, word_file):
    # Initialize the Converter with the PDF file
    cv = Converter(pdf_file)
    
    # Convert the PDF to Word (DOCX)
    cv.convert(word_file, start=0, end=None)
    
    # Close the Converter object
    cv.close()

# Example usage:
pdf_file = 'example.pdf'  # Replace with your PDF file path
word_file = 'output.docx'  # The desired output Word file

convert_pdf_to_word(pdf_file, word_file)
