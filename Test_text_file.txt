import pdfplumber

# Open the PDF file
with pdfplumber.open("your-file.pdf") as pdf:
    page = pdf.pages[0]
    
    # Manually define the table area (bounding box)
    # The coordinates (x0, top, x1, bottom) can be adjusted as per your table's location
    table_area = (50, 50, 550, 750)  # Example coordinates
    
    # Crop the page to the table area
    cropped_page = page.within_bbox(table_area)
    
    # Extract the table from the cropped area
    table = cropped_page.extract_table()
    
    for row in table:
        print(row)

    # Visualize the cropped area with debug mode
    im = page.within_bbox(table_area).to_image(resolution=300)
    im.debug_tablefinder()
