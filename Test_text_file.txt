import pandas as pd
import numpy as np

# Example DataFrame
data = {'Column1': [1, 2, 3],
        'Column2': [np.nan, 5, np.nan],
        'Column3': [4, np.nan, np.nan]}

df = pd.DataFrame(data)

# Create a new column based on conditions
df['NewColumn'] = np.where(df['Column2'].isna() & df['Column3'].isna(), 'Both NaN',
                           np.where(df['Column2'].isna(), 'Column2 NaN',
                                    np.where(df['Column3'].isna(), 'Column3 NaN', 'None NaN')))

print(df)
