import pdfplumber

def get_text_after_bbox(page, bbox):
    """
    Extracts text that comes after the bottom of a given bounding box until the end of the page.
    """
    bottom_text = ""
    for word in page.extract_words():
        word_bbox = (word['x0'], word['top'], word['x1'], word['bottom'])
        
        # Check if the word is below the bottom of the given bounding box
        if word_bbox[1] > bbox[3]:  # bbox[3] is the bottom edge of the bounding box
            bottom_text += word['text'] + " "

    return bottom_text.strip()

# Define the bounding box for your table
table_bbox = (366, 329, 469, 357)

# Open the PDF file
with pdfplumber.open("your-file.pdf") as pdf:
    for page in pdf.pages:
        # Extract the text after the given table's bounding box
        bottom_text = get_text_after_bbox(page, table_bbox)

        # Print the bottom text after the table's bounding box
        print(f"Text after the table on Page {page.page_number}:\n", bottom_text)
