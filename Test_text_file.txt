import pandas as pd

# Example DataFrame
data = {
    "ID": [1, 2, 3, 4, 5, 6],
    "Column1": ["Normal data", "cost if you use ...", "Another normal row", "Example", "cost if you use here", "More data"],
    "Column2": ["A", "B", "C", "D", "E", "F"],
    "Column3": [100, 200, 300, 400, 500, 600]
}
main_df = pd.DataFrame(data)

# String to search
search_string = "cost if you use"
replacement_text = "cost if you use the in-network provider"

# Replace rows with the specific text
main_df.loc[
    main_df["Column1"].str.contains(search_string, na=False), 
    ["Column1", "Column2", "Column3"]
] = replacement_text

print(main_df)
