import pdfplumber

# Path to your PDF file
pdf_path = "your_file.pdf"

# Open the PDF file with pdfplumber
with pdfplumber.open(pdf_path) as pdf:
    # Iterate through all pages in the PDF
    for i, page in enumerate(pdf.pages):
        # Extract the text from the page
        page_text = page.extract_text()
        
        # Extract tables from the page
        tables = page.extract_tables()

        # Check if there is text on the page
        if page_text:
            # If there are tables, find the text between the tables
            if tables:
                table_text = ""
                for table in tables:
                    table_start_index = page_text.find(str(table[0][0]))
                    table_end_index = page_text.find(str(table[-1][-1])) + len(str(table[-1][-1]))
                    table_text += page_text[:table_start_index] + page_text[table_end_index:]

                # If text is found between tables, print it
                if table_text.strip():
                    print(f"Page {i+1} of {len(pdf.pages)} - Text between tables:")
                    print(table_text)
                else:
                    print(f"Page {i+1} of {len(pdf.pages)} - no data")
            else:
                print(f"Page {i+1} of {len(pdf.pages)} - no data")
        else:
            print(f"Page {i+1} of {len(pdf.pages)} - no data")
