import pdfplumber

def get_bottom_text(text, last_table_bbox, page):
    """Extract text that comes after the last table on the page."""
    bottom_text = ""
    for word in page.extract_words():
        # Extract the y-coordinate of the word
        if word['top'] > last_table_bbox[3]:  # Compare with the bottom of the last table
            bottom_text += word['text'] + " "
    return bottom_text.strip()

# Open the PDF file
with pdfplumber.open("your-file.pdf") as pdf:
    for page in pdf.pages:
        # Find all tables on the page
        tables = page.find_tables()

        # Check if there are any tables on the page
        if tables:
            # Get the bounding box of the last table
            last_table_bbox = tables[-1].bbox

            # Extract the text after the last table
            bottom_text = get_bottom_text(page.extract_text(), last_table_bbox, page)

            # Print the bottom text after the last table
            print(f"Bottom Text Below Last Table on Page {page.page_number}:\n", bottom_text)
        else:
            # If no tables, you can decide what to do (e.g., extract the whole page text)
            print(f"No tables found on Page {page.page_number}.")
