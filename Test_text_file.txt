import fitz  # PyMuPDF

# Open the PDF
pdf_path = "your_pdf_with_highlighted_text.pdf"
doc = fitz.open(pdf_path)

# Iterate through the pages and remove highlight annotations
for page_num in range(len(doc)):
    page = doc.load_page(page_num)
    
    # Find and remove highlight annotations
    for annot in page.annots():
        if annot.type[0] == 8:  # Annotation type 8 corresponds to highlight
            page.delete_annot(annot)

# Save the modified PDF without highlights
output_pdf = "output_without_highlight.pdf"
doc.save(output_pdf)
doc.close()

print(f"Highlights removed. Saved to {output_pdf}")
