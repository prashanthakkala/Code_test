import pyodbc
import pandas as pd

# Establish connection to SQL Server
conn = pyodbc.connect(
    'DRIVER={SQL Server};'
    'SERVER=your_server_name;'
    'DATABASE=your_database_name;'
    'UID=your_username;'
    'PWD=your_password'
)

# Parameters for the stored procedure
param1_value = 'some_value'
param2_value = 1234

# Execute the stored procedure with parameters and load result into a DataFrame
df = pd.read_sql_query("{CALL your_procedure_name (?, ?)}", conn, params=(param1_value, param2_value))

# Close the connection
conn.close()

# Display the DataFrame
print(df)
