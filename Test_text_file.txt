import pdfplumber

def print_text_lines(pdf_path):
    with pdfplumber.open(pdf_path) as pdf:
        for i, page in enumerate(pdf.pages):
            # Extract text lines from the page
            lines = page.extract_text_lines()
            
            print(f"Page {i + 1} of {len(pdf.pages)}:")
            if lines:
                for line in lines:
                    print(line)
            else:
                print("No text lines found on this page")

# Path to your PDF file
pdf_path = "your_file.pdf"
print_text_lines(pdf_path)
