import pdfplumber

# Open the PDF file
with pdfplumber.open("your-file.pdf") as pdf:
    page = pdf.pages[0]
    
    # Define the bounding box for the table area
    table_area = (50, 50, 550, 750)  # Example coordinates for the table
    
    # Crop the page to the table area
    cropped_page = page.within_bbox(table_area)
    
    # Extract the table from the cropped area
    tables = cropped_page.find_tables(table_settings={
        "vertical_strategy": "lines",   # Use vertical lines for table structure
        "horizontal_strategy": "lines-strict" # Allow horizontal lines if present
    })
    
    if tables:
        last_table = tables[-1]  # Get the last table
        for row in last_table.extract():
            print(row)
    else:
        print("No tables found in the specified area.")
