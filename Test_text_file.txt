from flask import Flask, render_template
from your_script import main

app = Flask(__name__)

@app.route("/")
def index():
    return render_template("index.html")

@app.route("/run", methods=["POST"])
def run_code():
    result = main()   # only main() runs everything
    return render_template("index.html", result=result)

if __name__ == "__main__":
    app.run(debug=True)


<!DOCTYPE html>
<html>
<head>
    <title>Run Selenium Script</title>
</head>
<body>

    <h2>Run Full Automation</h2>

    <!-- Run main() only -->
    <form action="/run" method="POST">
        <button type="submit" style="padding:10px 20px; font-size:18px;">
            Run Automation
        </button>
    </form>

    {% if result %}
        <h3>Output:</h3>
        <div style="border:1px solid #ccc; padding:10px;">
            {{ result }}
        </div>
    {% endif %}

</body>
</html>
