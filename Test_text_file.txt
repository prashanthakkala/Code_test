import pdfplumber

# Open the PDF file
with pdfplumber.open("your-file.pdf") as pdf:
    page = pdf.pages[0]
    
    # Define bounding boxes for different tables if necessary
    table_areas = [
        (50, 50, 550, 300),  # Example coordinates for the first table
        (50, 350, 550, 750)  # Example coordinates for the second table
    ]
    
    # Extract tables from each defined area
    for bbox in table_areas:
        cropped_page = page.within_bbox(bbox)
        tables = cropped_page.find_tables()
        
        if tables:
            # Get the last table from this cropped area
            last_table = tables[-1]
            
            # Extract and print the last table's content
            for row in last_table.extract():
                print(row)
