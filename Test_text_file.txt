from docx import Document
from docx.shared import RGBColor

def remove_yellow_highlights(docx_path, output_path):
    # Open the Word document
    doc = Document(docx_path)
    
    # Iterate through all paragraphs in the document
    for para in doc.paragraphs:
        for run in para.runs:
            if run.font.highlight_color:
                # Remove the highlight if it's yellow
                # Yellow in Word is typically represented as 'yellow'
                if run.font.highlight_color == 5:  # 5 corresponds to yellow highlight
                    run.font.highlight_color = None  # Remove the highlight

    # Save the modified document
    doc.save(output_path)

# Example usage
docx_path = "your_file.docx"
output_path = "output_without_yellow_highlight.docx"
remove_yellow_highlights(docx_path, output_path)
