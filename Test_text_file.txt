import pdfplumber

with pdfplumber.open("your_file.pdf") as pdf:
    for page in pdf.pages:
        lines = page.lines
        if lines:
            print(f"Page {page.page_number} has {len(lines)} lines.")
            for line in lines:
                print(line)
        else:
            print(f"Page {page.page_number} has no lines.")
