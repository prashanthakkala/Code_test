import pdfplumber

def extract_text_with_lines(pdf_path):
    with pdfplumber.open(pdf_path) as pdf:
        for page in pdf.pages:
            # Extract horizontal lines from the page
            lines = [line for line in page.lines if line['width'] > line['height']]
            
            # Sort lines based on the y-position
            lines.sort(key=lambda x: x['y0'], reverse=True)
            
            # Extract text between the lines
            for i in range(len(lines) - 1):
                top = lines[i]['y0']
                bottom = lines[i + 1]['y0']
                
                # Define a cropping box to extract text within the region
                cropped_page = page.within_bbox((0, bottom, page.width, top))
                
                # Extract text from the cropped region
                text = cropped_page.extract_text()
                
                print(f"Text between line {i} and line {i + 1}:\n{text}\n{'-'*40}")

# Provide the path to your PDF file
extract_text_with_lines("your_file.pdf")
