import pandas as pd

# ---- MAIN DF ----
main_df = pd.DataFrame({
    'MsgID': [101, 102, 103, 104],
    'Name': ['A', 'B', 'C', 'D']
})

# ---- ADD NEW COLUMNS (EMPTY) ----
new_cols = ['Status', 'Score', 'Remarks']
for col in new_cols:
    main_df[col] = None    # or np.nan

# ---- FOR LOOP TO INSERT VALUES ----
for index, row in main_df.iterrows():

    # Example: generate values using loop logic
    status_val = "OK" if row['MsgID'] % 2 == 0 else "PENDING"
    score_val = row['MsgID'] * 10
    remarks_val = f"Processed {row['Name']}"

    # Insert into the new columns
    main_df.at[index, 'Status'] = status_val
    main_df.at[index, 'Score'] = score_val
    main_df.at[index, 'Remarks'] = remarks_val

print(main_df)
