from PyPDF2 import PdfReader, PdfWriter

# Open the original PDF
input_pdf_path = "path/to/original_pdf.pdf"
output_pdf_path = "path/to/output_pdf.pdf"

reader = PdfReader(input_pdf_path)
writer = PdfWriter()

# Extract the first 10 pages
for page_num in range(10):  # Pages 0-9 (i.e., first 10 pages)
    page = reader.pages[page_num]
    writer.add_page(page)

# Save the extracted pages to a new PDF
with open(output_pdf_path, "wb") as output_pdf:
    writer.write(output_pdf)

print(f"New PDF with 10 pages saved to {output_pdf_path}")
